  151.     {

  153.         $arSelect   = array();

  155.         $arWhere    = array();
  156.  
  157.         $strResultForm = $this->query->getResultForm();
  158.         $filterGen     = new SparqlEngineDb_FilterGenerator($this);
  159.         switch ($strResultForm) {
  160.             case 'construct':
  161.                 $arResultVars = $this->query->getConstructPatternVariables();
  162.                 break;
  163.             default:
  164.                 $arResultVars = $this->query->getResultVars();
  165.                 break;
  166.         }
  167.  
  168.         $this->nTableId                 = 0;
  169.         $this->nGraphPatternCount       = 0;
  170.         $this->nUnionCount              = 0;
  171.         $this->nUnionTriplePatternCount = 0;
  172.         $this->arUnionVarAssignments[0] = array();
  173.  
  174.         foreach ($this->query->getResultPart() as $graphPattern) {
  175.             if (self::isEmptyPattern($graphPattern)) {
  176.                 continue;
  177.             }
  178.             if ($graphPattern->getUnion() !== null) {
  179.                 ++$this->nUnionCount;
  180.                 $this->nUnionTriplePatternCount = 0;
  181.                 $this->nGraphPatternCount = 0;
  182.                 $this->arUnionVarAssignments[$this->nUnionCount] = array();
  183.             }
  184.             $this->nTriplePatternCount = 0;
  185.             $arTriplePattern = $graphPattern->getTriplePattern();
  186.             if ($arTriplePattern != null) {
  187.                 foreach ($arTriplePattern as $triplePattern) {
  188.                     list (
  189.                         $arSelect[$this->nUnionCount][],
  190.                         $arFrom  [$this->nUnionCount][],
  191.                         $arWhere [$this->nUnionCount][]
  192.                     ) =
  193.                         $this->getTripleSql(
  194.                             $triplePattern,
  195.                             $graphPattern,
  196.                             $arResultVars
  197.                         );
  198.                     ++$this->nTableId;
  199.                     ++$this->nTriplePatternCount;
  200.                     ++$this->nUnionTriplePatternCount;
  201.                 }
  202.             }
  203.             ++$this->nGraphPatternCount;
  204.  
  205.         }
  206.  
  207.         //constraints extra. needed, since OPTIONAL parts are put after
  208.         // the current pattern while the constraint already refers to variables
  209.         // defined in there
  210.         $this->nGraphPatternCount       = 0;
  211.         $this->nUnionCount              = 0;
  212.         foreach ($this->query->getResultPart() as $graphPattern) {
  213.             if ($graphPattern->getUnion() !== null) {
  214.                 ++$this->nUnionCount;
  215.             }
  216.             $arConstraints = $graphPattern->getConstraint();
  217.             if ($arConstraints != null) {
  218.                 foreach ($arConstraints as $constraint) {
  219.                     $arWhere[$this->nUnionCount][count($arWhere[$this->nUnionCount]) - 1]
  220.                      .= $filterGen->createFilterSql(
  221.                         $constraint->getTree(),
  222.                         $graphPattern->getOptional() !== null,
  223.                         $this->nUnionCount
  224.                     );
  225.                 }
  226.             }
  227.             ++$this->nGraphPatternCount;
  228.         }
  229.  
  230.         $arSelect    = $this->createEqualSelects($arSelect);
  231.         $arStrSelect = array();
  232.  
  233.         switch ($strResultForm) {
  234.             case 'construct':
  235.             case 'describe':
  236.                 $strSelectType = 'SELECT';
  237.             case 'select':
  238.             case 'select distinct':
  239.                 if (!isset($strSelectType)) {
  240.                     $strSelectType = $strResultForm;
  241.                 }
  242.                 foreach ($arSelect as $nUnionCount => $arSelectPart) {
  243.                     $arSelectPart = self::removeNull($arSelectPart);
  244.                     if (count($arSelectPart) == 0) {
  245.                         throw new SparqlEngineDb_SqlGeneratorException('No variable that could be returned.');
  246.                     }
  247.                     $arStrSelect[$nUnionCount] = strtoupper($strSelectType) . ' ' . implode(', '   , $arSelectPart);
  248.                 }
  249.                 break;
  250.  
  251.             case 'ask':
  252.             case 'count':
  253.                 $strSelect = 'SELECT COUNT(*) as count';
  254.                 break;
  255.  
  256.             default:
  257.                 throw new SparqlEngineDb_SqlGeneratorException('Unsupported query type "' . $strResultForm . '"');
  258.                 break;
  259.         }
  260.  
  261.         $arSqls = array();
  262.         foreach ($arSelect as $nUnionCount => $arSelectPart) {
  263.             $arSqls[] = array(
  264.                 'select'    => $arStrSelect[$nUnionCount],
  265.                 'from'      => ' FROM '  . implode(' '    , self::removeNull($arFrom[$nUnionCount])),
  266.                 'where'     => ' WHERE ' . self::fixWhere(
  267.                             implode(' '  , self::removeNull($arWhere[$nUnionCount]))
  268.                 )
  269.             );
  270.         }
  271.         return $arSqls;
  272.     }//function createSql()

  276.     /**
  277.     *   Creates some SQL statements from the given triple pattern
  278.     *   array.
  279.     *
  280.     *   @param QueryTriple  $triple                 Array containing subject, predicate and object
  281.     *   @param GraphPattern $graphPattern           Graph pattern object
  282.     *
  283.     *   @return array   Array consisting of on array and two string values:
  284.     *                    SELECT, FROM and WHERE part
  285.     */
  286.     function getTripleSql(QueryTriple $triple, GraphPattern $graphPattern, $arResultVars)
  287.     {
  288.         $arSelect  = array();
  289.         $strFrom    = null;
  290.         $strWhere   = null;
  291.         $strWhereEquality   = '';
  292.  
  293.         $subject    = $triple->getSubject();
  294.         $predicate  = $triple->getPredicate();
  295.         $object     = $triple->getObject();
  296.  
  297.         $arRefVars      = array();
  298.         $strTablePrefix = 't' . $this->nTableId;
  299.  
  300.         /**
  301.         *   SELECT part
  302.         *   We do select only the columns we need for variables
  303.         */
  304.         if (self::isVariable($subject)) {
  305.             if (isset($this->arUnionVarAssignments[$this->nUnionCount][$subject])) {
  306.                 //already selected -> add equality check
  307.                 $strWhereEquality .= ' AND ' . self::getSqlEqualityCondition(
  308.                                 array($strTablePrefix, 's'),
  309.                                 $this->arVarAssignments[$subject]
  310.                             );
  311.                 $this->arUsedVarTypes[$subject]['s'] = true;
  312.             } else {
  313.                 if (isset($this->arVarAssignments[$subject][0])) {
  314.                     $strTablePrefix = $this->arVarAssignments[$subject][0];
  315.                 }
  316.                 $this->arVarAssignments[$subject] = array($strTablePrefix, 's');
  317.                 $this->arUnionVarAssignments[$this->nUnionCount][$subject] = array($strTablePrefix, 's');
  318.                 $this->arUsedVarTypes[$subject]['s'] = true;
  319.                 if (self::isResultVar($subject, $arResultVars)) {
  320.                     //new variable that needs to be selected
  321.                     $arSelect[$subject] = array(
  322.                         $strTablePrefix . '.subject as "'    . $strTablePrefix . '.' . $this->getSqlVariableNameValue($subject) . '"',
  323.                         $strTablePrefix . '.subject_is as "' . $strTablePrefix . '.' . $this->getSqlVariableNameIs($subject) . '"'
  324.                     );
  325.                     if (isset($this->arUsedVarAssignments[$subject])) {
  326.                         $arRefVars[$subject] = $strTablePrefix . '.subject';
  327.                     } else {
  328.                         $this->arUsedVarAssignments[$subject] = $strTablePrefix . '.subject';
  329.                     }
  330.                 }
  331.             }
  332.         }
  333.  
  334.         if (self::isVariable($predicate)) {
  335.             if (isset($this->arUnionVarAssignments[$this->nUnionCount][$predicate])) {
  336.                 //already selected -> add equality check
  337.                 $strWhereEquality .= ' AND ' . self::getSqlEqualityCondition(
  338.                                 array($strTablePrefix, 'p'),
  339.                                 $this->arVarAssignments[$predicate]
  340.                             );
  341.                 $this->arUsedVarTypes[$predicate]['p'] = true;
  342.             } else {
  343.                 if (isset($this->arVarAssignments[$predicate][0])) {
  344.                     $strTablePrefix = $this->arVarAssignments[$predicate][0];
  345.                 }
  346.                 $this->arVarAssignments[$predicate] = array($strTablePrefix, 'p');
  347.                 $this->arUnionVarAssignments[$this->nUnionCount][$predicate] = array($strTablePrefix, 'p');
  348.                 $this->arUsedVarTypes[$predicate]['p'] = true;
  349.                 if (self::isResultVar($predicate, $arResultVars)) {
  350.                     $arSelect[$predicate] = array(
  351.                         $strTablePrefix . '.predicate as "' . $strTablePrefix . '.' . $this->getSqlVariableNameValue($predicate) . '"'
  352.                     );
  353.                     if (isset($this->arUsedVarAssignments[$predicate])) {
  354.                         $arRefVars[$predicate] = $strTablePrefix . '.predicate';
  355.                     } else {
  356.                         $this->arUsedVarAssignments[$predicate] = $strTablePrefix . '.predicate';
  357.                     }
  358.                 }
  359.             }
  360.         }
  361.  
  362.         if (self::isVariable($object)) {
  363.             if (isset($this->arUnionVarAssignments[$this->nUnionCount][$object])) {
  364.                 //already selected -> add equality check
  365.                 $strWhereEquality .= ' AND ' . self::getSqlEqualityCondition(
  366.                                 array($strTablePrefix, 'o'),
  367.                                 $this->arVarAssignments[$object]
  368.                             );
  369.                 $this->arUsedVarTypes[$object]['o'] = true;
  370.             } else {
  371.                 if (isset($this->arVarAssignments[$object][0])) {
  372.                     $strTablePrefix = $this->arVarAssignments[$object][0];
  373.                 }
  374.                 $this->arVarAssignments[$object] = array($strTablePrefix, 'o');
  375.                 $this->arUnionVarAssignments[$this->nUnionCount][$object] = array($strTablePrefix, 'o');
  376.                 $this->arUsedVarTypes[$object]['o'] = true;
  377.                 if (self::isResultVar($object, $arResultVars)) {
  378.                     $arSelect[$object] = array(
  379.                         $strTablePrefix . '.object as "'     . $strTablePrefix . '.' . $this->getSqlVariableNameValue($object) . '"',
  380.                         $strTablePrefix . '.object_is as "'  . $strTablePrefix . '.' . $this->getSqlVariableNameIs($object) . '"',
  381.                         $strTablePrefix . '.l_language as "' . $strTablePrefix . '.' . $this->getSqlVariableNameLanguage($object) . '"',
  382.                         $strTablePrefix . '.l_datatype as "' . $strTablePrefix . '.' . $this->getSqlVariableNameDatatype($object) . '"',
  383.                     );
  384.                     if (isset($this->arUsedVarAssignments[$object])) {
  385.                         $arRefVars[$object] = $strTablePrefix . '.object';
  386.                     } else {
  387.                         $this->arUsedVarAssignments[$object] = $strTablePrefix . '.object';
  388.                     }
  389.                 }
  390.                 if (isset($this->query->varLanguages[$object])
  391.                  && $this->query->varLanguages[$object] !== null
  392.                 ) {
  393.                     $strWhereEquality .=
  394.                         ' AND ' . $strTablePrefix . '.l_language = "'
  395.                         . addslashes($this->query->varLanguages[$object]) . '"';
  396.                 }
  397.                 if (isset($this->query->varDatatypes[$object])
  398.                  && $this->query->varDatatypes[$object] !== null
  399.                 ) {
  400.                     $strWhereEquality .=
  401.                         ' AND ' . $strTablePrefix . '.l_datatype = "'
  402.                         . addslashes($this->query->varDatatypes[$object]) . '"';
  403.                 }
  404.             }
  405.         }
  406.  
  407.         /**
  408.         *   FROM part
  409.         */
  410.         if ($this->nUnionTriplePatternCount == 0) {
  411.             //first FROM
  412.             $strFrom    = 'statements as ' . $strTablePrefix;
  413.         } else {
  414.             //normal join
  415.             $strFrom    = 'LEFT JOIN statements as ' . $strTablePrefix
  416.                         . ' ON t0.modelID = ' . $strTablePrefix . '.modelID';
  417.             foreach ($arRefVars as $strRefVar => $strSqlVar) {
  418.                 $strFrom .= ' AND ' . $this->arUsedVarAssignments[$strRefVar] . ' = ' . $strSqlVar;
  419.             }
  420.  
  421.             if ($graphPattern->getOptional() !== null) {
  422.                 $strFrom .=  $this->getSqlCondition($subject  , $strTablePrefix, 'subject')
  423.                            . $this->getSqlCondition($predicate, $strTablePrefix, 'predicate')
  424.                            . $this->getSqlCondition($object   , $strTablePrefix, 'object')
  425.                            . $strWhereEquality;
  426.             }
  427.         }
  428.  
  429.  
  430.         /**
  431.         *   WHERE part
  432.         */
  433.         if ($this->nUnionTriplePatternCount == 0) {
  434.             $strWhere  = $strTablePrefix . '.modelID = ' . intval($this->modelID);
  435.         }
  436.         if ($graphPattern->getOptional() === null || $this->nGraphPatternCount == 0) {
  437.             $strWhere .=  $this->getSqlCondition($subject  , $strTablePrefix, 'subject')
  438.                         . $this->getSqlCondition($predicate, $strTablePrefix, 'predicate')
  439.                         . $this->getSqlCondition($object   , $strTablePrefix, 'object')
  440.                         . $strWhereEquality;
  441.         }
  442.  
  443.         return array($arSelect, $strFrom, $strWhere);
  444.     }//function getTripleSql(QueryTriple $triple)

  448.     /**
  449.     *   Creates SELECT statements that have the same number of columns.

  451.     *
  452.     *   @param array $arSelect  Array of arrays.
  453.     *        array(
  454.     *            //for each union part one
  455.     *            0 => array(
  456.     *                //foreach triple pattern
  457.     *                0 => array(
  458.     *                    '?person'   => array(
  459.     *                        't0.subject as "t0.subject"'
  460.     *                    )
  461.     *                )
  462.     *            )
  463.     *        )
  464.     *   @return array Array of SELECT strings
  465.     */
  466.     protected function createEqualSelects($arSelect)
  467.     {
  468.         $arNewSelect = array();
  469.         if (count($arSelect) == 1) {
  470.             foreach ($arSelect[0] as $arTripleVars) {
  471.                 $ar = array();
  472.                 foreach ($arTripleVars as $arVarParts) {
  473.                     $ar[] = implode(', ', $arVarParts);
  474.                 }
  475.                 if (count($ar) > 0) {
  476.                     $arNewSelect[0][] = implode(', ', $ar);
  477.                 }
  478.             }
  479.             return $arNewSelect;
  480.         }
  481.  
  482.         $arVars = array();
  483.          foreach ($arSelect as $arUnionVars) {
  484.             foreach ($arUnionVars as $arTripleVars) {
  485.                 $arVars = array_merge($arVars, array_keys($arTripleVars));
  486.             }
  487.         }
  488.         $arVars = array_unique($arVars);
  489.  
  490.         foreach ($arSelect as $nUnionCount => $arUnionVars) {
  491.             $arSelectVars = array();
  492.             foreach ($arUnionVars as $arTripleVars) {
  493.                 foreach ($arTripleVars as $strVar => $arVarParts) {
  494.                     $arSelectVars[$strVar] = $arVarParts;
  495.                 }
  496.             }
  497.  
  498.             $ars = array();
  499.             foreach ($arVars as $strVar) {
  500.                 if (isset($arSelectVars[$strVar])) {
  501.                     $ar     = $arSelectVars[$strVar];
  502.                     $nCount = count($arSelectVars[$strVar]);
  503.                 } else {
  504.                     $ar     = array();
  505.                     $nCount = 0;
  506.                 }
  507.  
  508.                 if ($nCount == 0) {
  509.                     //nothing of this variable in this union part
  510.                     $ar[] = 'NULL as '
  511.                         . '"' . $this->arVarAssignments[$strVar][0] . '.' . $this->arVarAssignments[$strVar]['sql_value'] . '"';
  512.                 }
  513.                 if ((
  514.                     isset($this->arUsedVarTypes[$strVar]['o'])
  515.                     || isset($this->arUsedVarTypes[$strVar]['s'])
  516.                     ) && $nCount < 2
  517.                 ) {
  518.                     //it's a subject or object, but we don't want the type
  519.                     $ar[] = 'NULL as '
  520.                         . '"' . $this->arVarAssignments[$strVar][0] . '.' . $this->arVarAssignments[$strVar]['sql_is'] . '"';
  521.                 }
  522.                 if (isset($this->arUsedVarTypes[$strVar]['o']) && $nCount < 4) {
  523.                     //it's a subject or object, but we don't want the type
  524.                     if (isset($this->arVarAssignments[$strVar]['sql_lang'])) {
  525.                         $strColLanguage = $this->arVarAssignments[$strVar]['sql_lang'];
  526.                     } else {
  527.                         $strColLanguage = 'dummyLang';
  528.                     }
  529.                     if (isset($this->arVarAssignments[$strVar]['sql_type'])) {
  530.                         $strColDatatype = $this->arVarAssignments[$strVar]['sql_type'];
  531.                     } else {
  532.                         $strColDatatype = 'dummyType';
  533.                     }
  534.                     $ar[] = 'NULL as '
  535.                         . '"' . $this->arVarAssignments[$strVar][0] . '.' . $strColLanguage . '"';
  536.                     $ar[] = 'NULL as '
  537.                         . '"' . $this->arVarAssignments[$strVar][0] . '.' . $strColDatatype . '"';
  538.                 }
  539.                 $ars[] = implode(', ', $ar);
  540.             }
  541.             $arNewSelect[$nUnionCount] = $ars;
  542.         }
  543.  
  544.         return $arNewSelect;
  545.     }//protected function createEqualSelects($arSelect)

  549.     /**
  550.     *   Creates an SQL statement that checks for the value
  551.     *   of some subject/predicate/object
  552.     *
  553.     *   @param mixed    $bject          subject|predicate|object
  554.     *   @param string   $strTablePrefix Table prefix (e.g. "t0")
  555.     *   @param string   $strType        Type of $bject ('subject'|'predicate'|'object')
  556.     *   @return string  Part of the SQL query (prefixed with AND)
  557.     */
  558.     function getSqlCondition($bject, $strTablePrefix, $strType)
  559.     {
  560.         if (is_string($bject)) {
  561.             if (self::isVariable($bject)) {
  562.                 //variable?
  563.                 if (self::isPreparedVariable($bject)) {
  564.                     //no, not really
  565.                     $value = $this->getPreparedVariablePlaceholder($bject);
  566.                 } else {
  567.                     //yes
  568.                     return null;
  569.                 }
  570.             } else {
  571.                 $value = $this->dbConn->qstr($bject);
  572.             }
  573.             //literal
  574.             return ' AND ' . $strTablePrefix . '.' . $strType . ' = ' . $value;
  575.         }
  576.  
  577.         if ($bject instanceof BlankNode) {
  578.             //Blank node
  579.             throw new SparqlEngineDb_SqlGeneratorException(
  580.                 'FIXME: Querying for blank nodes not supported'
  581.             );
  582.  
  583.         } else if ($bject instanceof Resource) {
  584.             //Resource
  585.             $r = ' AND ' . $strTablePrefix . '.' . $strType . ' = '
  586.                 . $this->dbConn->qstr($bject->getURI());
  587.             if ($strType !== 'predicate') {
  588.                 $r .= ' AND ' . $strTablePrefix . '.' . $strType . '_is ='
  589.                 . ' "r"';
  590.             }
  591.             return $r;
  592.  
  593.         } else if ($bject instanceof Literal) {
  594.             //Literal
  595.             //I'm doubling Filter code here, but what the hell
  596.             $strColDatatype = $strTablePrefix . '.l_datatype';
  597.             if ($bject->dtype == 'http://www.w3.org/2001/XMLSchema#integer'
  598.              || $bject->dtype == 'http://www.w3.org/2001/XMLSchema#double'
  599.             ) {
  600.                 $strVariable = 'CAST(' . $strTablePrefix . '.' . $strType . ' AS DECIMAL)';
  601.                 $strValue    = $bject->getLabel();
  602.             } else {
  603.                 $strVariable = $strTablePrefix . '.' . $strType;
  604.                 $strValue    = $this->dbConn->qstr($bject->getLabel());
  605.             }
  606.             $r = ' AND ' . $strVariable . ' = ' . $strValue;
  607.             if ($strType !== 'predicate') {
  608.                 $r .= ' AND ' . $strTablePrefix . '.' . $strType . '_is ='
  609.                 . ' "l"';
  610.             }
  611.  
  612.             if ($strType == 'object') {
  613.                 if ($bject->dtype == '' || $bject->dtype == 'http://www.w3.org/2001/XMLSchema#string') {
  614.                     //string
  615.                     $r .= ' AND ('
  616.                         . $strColDatatype . ' = ""'
  617.                         . ' OR ' . $strColDatatype . ' = "http://www.w3.org/2001/XMLSchema#string"'
  618.                         . ')';
  619.                 } else {
  620.                     $r .= ' AND ' . $strColDatatype . ' = "'
  621.                         . $bject->dtype
  622.                         . '"';
  623.                 }
  624.             }
  625.  
  626.             if ($bject->lang != '') {
  627.                 $strColLanguage = $strTablePrefix . '.l_language';
  628.                 $r .= ' AND ' . $strColLanguage . ' = '
  629.                    . $this->dbConn->qstr($bject->lang);
  630.             }
  631.             return $r;
  632.  
  633.         } else {
  634.             throw new SparqlEngineDb_SqlGeneratorException(
  635.                 'Unsupported sentence part: ' . get_class($bject)
  636.             );
  637.         }
  638.     }//function getSqlCondition($bject, $strTablePrefix, $strType)

  642.     /**
  643.     *   Checks if the sentence part (subject, predicate or object) in
  644.     *   $arNew has the same content as $arOld.
  645.     *   Required for queries like ":x ?a ?a" where predicate and object
  646.     *   need to have the same value
  647.     *
  648.     *   @param array    $arNew  array($strTablePrefix, $strType = s|p|o)
  649.     *   @param array    $arOld  array($strTablePrefix, $strType = s|p|o)
  650.     *   @return string 
  651.     */
  652.     protected static function getSqlEqualityCondition($arNew, $arOld)
  653.     {
  654.         $chTypeNew         = $arNew[1]; $chTypeOld         = $arOld[1];
  655.         $strTablePrefixNew = $arNew[0]; $strTablePrefixOld = $arOld[0];
  656.  
  657.         if ($chTypeNew == 'p' || $chTypeOld == 'p') {
  658.             //just check value
  659.             //FIXME: it might be I need to check for resource type in object and subject
  660.             return
  661.                   $strTablePrefixNew . '.' . self::$arTableColumnNames[$chTypeNew]['value']
  662.                 . ' = '
  663.                 . $strTablePrefixOld . '.' . self::$arTableColumnNames[$chTypeOld]['value']
  664.                 ;
  665.         } else if ($chTypeNew == 's' || $chTypeOld == 's') {
  666.             //check value and type
  667.             return
  668.                   $strTablePrefixNew . '.' . self::$arTableColumnNames[$chTypeNew]['value']
  669.                 . ' = '
  670.                 . $strTablePrefixOld . '.' . self::$arTableColumnNames[$chTypeOld]['value']
  671.                 . ' AND '
  672.                 . $strTablePrefixNew . '.' . self::$arTableColumnNames[$chTypeNew]['is']
  673.                 . ' = '
  674.                 . $strTablePrefixOld . '.' . self::$arTableColumnNames[$chTypeOld]['is']
  675.                 ;
  676.         } else {
  677.             //two objects -> check everything
  678.             return
  679.                   $strTablePrefixNew . '.object = '     . $strTablePrefixOld . '.object'
  680.                 . ' AND '
  681.                 . $strTablePrefixNew . '.object_is = '  . $strTablePrefixOld . '.object_is'
  682.                 . ' AND '
  683.                 . $strTablePrefixNew . '.l_language = ' . $strTablePrefixOld . '.l_language'
  684.                 . ' AND '
  685.                 . $strTablePrefixNew . '.l_datatype = ' . $strTablePrefixOld . '.l_datatype'
  686.                 ;
  687.         }

  829.     /**

  833.     protected function fixWhere($strWhere)
  834.     {
  835.         $strWhere = ltrim($strWhere);
  836.         if (substr($strWhere, 0, 4) == 'AND ') {
  837.             $strWhere = substr($strWhere, 4);
