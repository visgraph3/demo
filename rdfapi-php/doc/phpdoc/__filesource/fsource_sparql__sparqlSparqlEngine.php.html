  152.     protected function matchPattern(&$patternlist, $graphlist, &$graphPattern) {
  153.         // generate an empty result set
  154.         $finalRes = null;
  155.         // if the GraphPattern has triple patterns
  156.         if($graphPattern->getTriplePattern()>0){
  157.             // check if the pattern has a GRAPH clause and if this Iri is in $graphlist
  158.             $newGraphList = $this->_checkGraphs($graphPattern,$graphlist);
  159.             if($newGraphList){
  160.                 $qt = $graphPattern->getTriplePattern();
  161.                 $resultSet = $this->findTuplesMatchingOnePattern($qt[0], $newGraphList);
  162.                 for ($i=1; $i<count($qt); $i++) {
  163.                     $rs = $this->findTuplesMatchingOnePattern($qt[$i], $newGraphList);
  164.                     $resultSet = $this->joinTuples($resultSet, $rs);
  165.                     if(!$resultSet)
  166.                     break;
  167.                 }
  168.                 if($finalRes != null){
  169.                     $finalRes = $this->joinTuples($finalRes,$resultSet);
  170.                 }else{
  171.                     $finalRes = $resultSet;
  172.                 }
  173.             }
  174.         }
  175.         // dependencies between pattern results
  176.         $patternlist[$graphPattern->getId()]['hasOptional']     = 0;
  177.         $patternlist[$graphPattern->getId()]['hasUnion']        = 0;
  178.         $patternlist[$graphPattern->getId()]['patternResult']   = $finalRes;
  179.  
  180.         $op = $graphPattern->getOptional();
  181.         $un = $graphPattern->getUnion();
  182.  
  183.         $patternlist[$graphPattern->getId()]['optionalTo']      = $op;
  184.         if(is_int($op))
  185.         $patternlist[$op]['hasOptional']++;
  186.  
  187.         $patternlist[$graphPattern->getId()]['unionWith']       = $un;
  188.         if(is_int($un))
  189.         $patternlist[$un]['hasUnion']++;
  190.  
  191.         $constraint = $graphPattern->getConstraint();
  192.         if($constraint != null){
  193.             foreach($constraint as $constr){
  194.                 if($constr->isOuterFilter()){
  195.                     $patternlist[$graphPattern->getId()]['outerFilter'][]          = $constr;
  196.                     $patternlist[$graphPattern->getId()]['innerFilter'][]          = null;
  197.                 }else{
  198.                     $patternlist[$graphPattern->getId()]['innerFilter'][]          = $constr;
  199.                     $patternlist[$graphPattern->getId()]['outerFilter'][]          = null;
  200.                 }
  201.             }
  202.         }else{
  203.             $patternlist[$graphPattern->getId()]['innerFilter']          = null;
  204.             $patternlist[$graphPattern->getId()]['outerFilter']          = null;
  205.         }
  206.     }

  210.     * Finds Tuples matching one TriplePattern.

  216.     protected function findTuplesMatchingOnePattern($pattern, $graphlist){
  217.         $var = null;
  218.         $sub  = $pattern->getSubject();
  219.         $pred = $pattern->getPredicate();
  220.         $obj  = $pattern->getObject();
  221.  
  222.         if(is_string($sub)||$sub instanceof BlankNode){
  223.             if(is_string($sub))
  224.             $var['sub'] = $sub;
  225.             $sub = null;
  226.         }
  227.         if(is_string($pred)||$pred instanceof BlankNode ){
  228.             if(is_string($pred))
  229.             $var['pred'] = $pred;
  230.             $pred = null;
  231.         }
  232.         if(is_string($obj)||$obj instanceof BlankNode){
  233.             if(is_string($obj))
  234.             $var['obj'] = $obj;
  235.             $obj = null;
  236.         }
  237.         $intBindings = $this->_buildIntBindings($var);
  238.         $k = 0;
  239.  
  240.         $key = 0;
  241.         // search in named graphs
  242.         if($graphlist['var'][0] != null||$graphlist['list'][0] != null){
  243.             foreach($graphlist['list'] as $key => $graphnode){
  244.  
  245.                 // query the dataset
  246.                 $it = $this->dataset->findInNamedGraphs($graphnode,$sub,$pred,$obj,false);
  247.                 if($it->valid()){
  248.                     // add statements to the result list
  249.                     while($it->valid()){
  250.                         if($graphnode == null){
  251.                             $element = $it->current()->getStatement();
  252.                             $grname  = $it->current()->getGraphname();
  253.                         }else{
  254.                             if($it->current() instanceof Quad)
  255.                                 $element = $it->current()->getStatement();
  256.                             else
  257.                                 $element = $it->current();
  258.  
  259.                             $grname  = $graphnode;
  260.                         }
  261.                         if($this->checkIntBindings($element,$intBindings)){
  262.                             $resmodel['trip'][$k]  = $element;
  263.                             $resmodel['graph'][$k] = $grname;
  264.                         //    $resmodel['graphvar'][$k] = $graphlist['var'][$key];
  265.                             $resmodel['graphvar'][$k] = $graphlist['var'][0];
  266.                             $k++;
  267.  
  268.                         }
  269.                         $it->next();
  270.                     }
  271.                 }
  272.  
  273.             }
  274.         }
  275.         // search in the default graph
  276.         if($graphlist['list'][0] == null && $graphlist['var'][0] == null){
  277.  
  278.  
  279.             $gr = $this->dataset->getDefaultGraph();
  280.  
  281.             $res = $gr->find($sub,$pred,$obj);
  282.  
  283.             foreach($res->triples as $innerkey => $element){
  284.                 if($this->checkIntBindings($element,$intBindings)){
  285.                         $resmodel['trip'][$k]  = $element;
  286.                         $resmodel['graph'][$k] = null;
  287.                         $resmodel['graphvar'][$k] = $graphlist['var'][$key];
  288.                         $k++;
  289.                     }
  290.             }
  291.         }
  292.         if($k == 0)
  293.         return false;
  294.         return $this->_buildResultSet($pattern,$resmodel);
  295.     }

  297.     /**
  298.     * Checks it there are internal bindings between variables.
  299.     *

  301.     * @param  Array   $intBindings 
  302.     * @return boolean 
  303.     */
  304.     protected function checkIntBindings($trip, $intBindings){
  305.         switch($intBindings){
  306.             case -1:
  307.             return true;
  308.             break;
  309.             case 0:
  310.             if($trip->subj != $trip->pred)
  311.             return false;
  312.             break;
  313.             case 1:
  314.             if(is_a($trip->obj,'Literal'))
  315.             return false;
  316.             if($trip->subj != $trip->obj)
  317.             return false;
  318.             break;
  319.             case 2:
  320.             if(is_a($trip->obj,'Literal'))
  321.             return false;
  322.             if($trip->pred != $trip->obj)
  323.             return false;
  324.             break;
  325.             case 3:
  326.             if(is_a($trip->obj,'Literal'))
  327.             return false;
  328.             if($trip->pred != $trip->obj || $trip->pred != $trip->subj )
  329.             return false;
  330.             break;
  331.         }
  332.         return true;

  336.     /**
  337.     * Perform an SQL-like inner join on two resultSets.
  338.     *
  339.     * @param   Array   &$finalRes 
  340.     * @param   Array   &$res 
  341.     * @return  Array 
  342.     */
  343.     protected function joinTuples(&$finalRes, &$res) {
  344.  
  345.         if (!$finalRes || !$res)
  346.         return array();
  347.  
  348.         // find joint variables and new variables to be added to $finalRes
  349.         $jointVars = array();
  350.         $newVars = array();
  351.         $k = key($res);
  352.  
  353.         foreach ($res[$k] as $varname => $node) {
  354.             if (array_key_exists($varname, $finalRes[0]))
  355.             $jointVars[] = $varname;
  356.             else
  357.             $newVars[] = $varname;
  358.         }
  359.  
  360.         // eliminate rows of $finalRes in which the values of $jointVars do not have
  361.         // a corresponding row in $res.
  362.         foreach ($finalRes as $n => $fRes) {
  363.             foreach ($res as $i => $r) {
  364.                 $ok = TRUE;
  365.                 foreach ($jointVars as $j_varname)
  366.                 if ($r[$j_varname] != $fRes[$j_varname]) {
  367.                     $ok = FALSE;
  368.                     break;
  369.                 }
  370.                 if ($ok)
  371.                 break;
  372.             }
  373.             if (!$ok)
  374.             unset($finalRes[$n]);
  375.         }
  376.  
  377.         // join $res and $finalRes
  378.         $joinedRes = array();
  379.         foreach ($res as $r) {
  380.             foreach ($finalRes as $n => $fRes) {
  381.                 $ok = TRUE;
  382.                 foreach ($jointVars as $j_varname)
  383.                 if ($r[$j_varname] != $fRes[$j_varname]) {
  384.                     $ok = FALSE;
  385.                     break;
  386.                 }
  387.                 if ($ok) {
  388.                     $joinedRow = $finalRes[$n];
  389.                     foreach($newVars as $n_varname)
  390.                     $joinedRow[$n_varname] = $r[$n_varname];
  391.                     $joinedRes[] = $joinedRow;
  392.                 }
  393.             }
  394.         }
  395.         return $joinedRes;
  396.     }

  399.     /**
  400.     * Joins OPTIONAL pattern results.
  401.     *
  402.     * @param   Array   &$finalRes 
  403.     * @param   Array   &$res 
  404.     * @return  Array    the joined Array
  405.     */
  406.     protected function joinOptionalTuples(&$finalRes, &$res) {
  407.  
  408.         if(!$finalRes && !$res)
  409.         return array();
  410.  
  411.         if(!$finalRes)
  412.         return $res;
  413.  
  414.         if(!$res)
  415.         return $finalRes;
  416.  
  417.         // find joint variables and new variables to be added to $finalRes
  418.         $jointVars = array();
  419.         $newVars = array();
  420.         $result = array();
  421.  
  422.         $k = key($res);
  423.  
  424.         foreach ($res[$k] as $varname => $node) {
  425.             if (array_key_exists($varname, $finalRes[0])){
  426.                 $jointVars[] = $varname;
  427.             }else{
  428.                 $newVars[] = $varname;
  429.             }
  430.         }
  431.         $joined = array();
  432.         $joinc = 0;
  433.         foreach($finalRes as $i =>$fRes){
  434.             foreach($res as $n =>$r){
  435.                 $join = false;
  436.                 foreach($jointVars as $j_varname){
  437.                     if($r[$j_varname]==$fRes[$j_varname]){
  438.                         $join = true;
  439.                         //break;
  440.                     }else{
  441.                         $join = false;
  442.                     }
  443.                 }
  444.                 if($join){
  445.                     $result[$joinc] = $fRes;
  446.                     foreach($newVars as $n_varname)
  447.                     $result[$joinc][$n_varname] = $r[$n_varname];
  448.                     $joined[]=$n;
  449.                     $joinc++;
  450.                 }
  451.  
  452.             }
  453.         }
  454.  
  455.         $count = count($result);
  456.         foreach($res as $k =>$val){
  457.             if(!in_array($k,$joined)){
  458.                 $result[$count] = $finalRes[0];
  459.                 foreach($result[$count] as $varname => $varVal){
  460.                     $result[$count][$varname]='';
  461.                 }
  462.  
  463.                 foreach($val as $varname2 => $varVal2){
  464.                     $result[$count][$varname2]=$varVal2;
  465.                 }
  466.                 $count++;

  749.     /**
